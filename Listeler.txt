public String[] satirBilgisiAl(int secilenSatir)
        {
            int sutunSayisi=jTable1.getColumnCount();
            String[] result =new String[sutunSayisi];
            for (int i = 0; i < sutunSayisi; i++) {
                result[i]=jTable1.getModel().getValueAt(secilenSatir, i).toString();
            }
            return result;
        }
private void MusterileriYukle()  
        {
            try
            {
                Baglanti baglanti = new Baglanti();
                Connection conn = baglanti.baglan();
                try
                {
                    Statement st=conn.createStatement();
                    ResultSet rs = st.executeQuery("Select * from "+"Musteriler");
                    DefaultTableModel DTM = new DefaultTableModel();
                    jTable1.setModel(DTM);
                    DTM.addColumn("Tc");
                    DTM.addColumn("Ad Soyad");
                    DTM.addColumn("Doğum Tarihi");
                    DTM.addColumn("Doğum Yeri");
                    DTM.addColumn("Cinsiyet");
                    DTM.addColumn("Adres");
                    DTM.addColumn("Telefon");
                    DTM.addColumn("Eposta");
                    
                    while(rs.next())
                    {
                        DTM.addRow(new String[]{rs.getString("Tc"),rs.getString("AdSoyad"),rs.getString("DogumTarihi"),rs.getString("DogumYeri"),rs.getString("Cinsiyet"),rs.getString("Adres"),rs.getString("Tel"),rs.getString("Eposta")});
                        
                    }
                    rs.close();
                    st.close();
                    conn.close();
                    jLabel1.setText("Müşteri Tc: ");
                    
                }
                catch(Exception exp)
                {
                    JOptionPane.showMessageDialog(rootPane, exp.getMessage());
                }
            }
            catch(Exception ex)
            {
                Logger.getLogger(Listeler.class.getName()).log(Level.SEVERE, null,ex);
            }
        }
